S -> IMPORT_STATEMENT PROGRAM
PROGRAM -> ENDLINE IMPORT_STATEMENT
IMPORT_STATEMENT -> IMPORT_FROM IMPORT_CONTENT_AS | IMPORT_CONTENT_AS | IMPORT_FROM IMPORT_CONTENT
IMPORT_CONTENT_AS -> IMPORT_CONTENT IMPORT_AS
IMPORT_FROM -> FROM OBJECT
IMPORT_CONTENT -> IMPORT OBJECT
IMPORT_AS -> AS OBJECT
FROM -> from
OBJECT -> object
IMPORT -> import
AS -> as
ENDLINE -> \n


// DEF
S -> DEF_STATEMENT ENDLINE DEF_BLOCK
DEF_STATEMENT -> DEF OBJECT DEF_PARAM
DEF_PARAM -> DEF_CONTENT COLON
DEF_CONTENT -> PARENTHESES | OPARENTHESES OBJECT CPARENTHESES
PARENTHESES -> OPARENTHESES CPARENTHESES
DEF_BLOCK -> OBJECT PARENTHESES
DEF -> def
OPARENTHESES -> (
CPARENTHESES -> )
COLON -> :
ENDLINE -> \n
